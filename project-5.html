SAMPLE IMPLEMENTATION

ESPHome Configuration

Standard configuration for a lab room featuring light control with physical switch override and automated door lock.

esphome:
  name: lab_controller
  platform: ESP32
  board: esp32dev

# Connectivity
networking:
  wifi:
    ssid: "Your_SSID"
    password: "Your_Password"

# Light Control with Physical Switch Synchronization
binary_sensor:
  - platform: gpio
    pin:
      number: GPIO14
      mode: INPUT_PULLUP
    name: "Lab Wall Switch"
    on_press:
      - light.toggle: lab_main_light

  - platform: gpio
    pin:
      number: GPIO12
      mode: INPUT_PULLUP
    name: "Lab Door Status"
    device_class: door

# Outputs
switch:
  - platform: gpio
    pin: GPIO13
    name: "Lab Door Lock"
    inverted: true # For fail-safe electronic locks

light:
  - platform: binary
    name: "Lab Main Light"
    output: light_output

output:
  - platform: gpio
    pin: GPIO27
    id: light_output

# Local Automation: Motion-based power saving
sensor:
  - platform: gpio
    pin: GPIO33
    name: "Lab Motion Sensor"
    device_class: motion
    on_state:
      if:
        condition:
          binary_sensor.is_off: lab_motion_sensor
        then:
          - delay: 5min
          - light.turn_off: lab_main_light
